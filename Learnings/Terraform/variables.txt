1. String

    Use case:
        EC2 instance name.

        variables.tf
            variable "instance_name" {
                type = string
            }

        terraform.tfvars
            instance_name = "web-server"

        main.tf
            resource "aws_instance" "web" {
                ami = "ami-123456"
                instance_type = "t2.micro"
                tags = {
                    Name = var.instance_name
                }
            }

        Result: EC2 Name = web-server

2. Number
    Use case: Number of EC2 instances.

    variables.tf
        variable "instance_count" {
            type = number
        }

    terraform.tfvars
        instance_count = 3

    main.tf
        resource "aws_instance" "web" {
            count = var.instance_count
            ami = "ami-123456"
            instance_type = "t2.micro"
        }
    
3. Bool
    Use case: Enable/disable monitoring.

    variables.tf
        variable "enable_monitoring" {
            type = bool
        }

    terraform.tfvars
        enable_monitoring = true

    main.tf
        resource "aws_instance" "web" {
            ami = "ami-123456"
            instance_type = "t2.micro"
            monitoring = var.enable_monitoring
        }

4. List
    A list stores multiple values of the same type.

    Example: Multiple availability zones.

        variables.tf
            variable "availability_zones" {
                type = list(string)
            }

        terraform.tfvars
            availability_zones = [
                "us-east-1a",
                "us-east-1b",
                "us-east-1c"
            ]

        main.tf
            resource "aws_subnet" "private" {
                count = length(var.availability_zones)
                vpc_id = aws_vpc.main.id
                availability_zone = var.availability_zones[count.index]
            }

        Creates:
            Subnet-1 → us-east-1a
            Subnet-2 → us-east-1b
            Subnet-3 → us-east-1c

5. Map
    Map stores key-value pairs.

    Example:
        Environment → Instance type.

        variables.tf
            variable "instance_types" {
                type = map(string)
            }

        terraform.tfvars
            instance_types = {
                dev  = "t2.micro"
                qa   = "t3.small"
                prod = "m5.large"
            }

        main.tf
            resource "aws_instance" "web" {
                ami = "ami-123456"
                instance_type = var.instance_types[var.environment]
            }

        If: environment = "prod"
        Terraform gets: m5.large
    
6. Object
    
    Object stores multiple attributes with different types.

    Example: EC2 configuration.

        variables.tf
            variable "server_config" {
                type = object({
                    name          = string
                    instance_type = string
                    disk_size     = number
                    monitoring    = bool
                })
            }

        terraform.tfvars
            server_config = {
                name          = "web"
                instance_type = "t3.medium"
                disk_size     = 50
                monitoring    = true
            }

        main.tf
            resource "aws_instance" "web" {
                ami = "ami-123456"
                instance_type = var.server_config.instance_type
                monitoring = var.server_config.monitoring
                tags = {
                    Name = var.server_config.name
                }
            }

7. Tuple
    Tuple is like a list, but: order matters, each element can have different types

    Example: ["web",3,true]

    Types:
        index 0 → string
        index 1 → number
        index 2 → bool

    variables.tf
        variable "server_details" {
            type = tuple([
                string,
                number,
                bool
            ])
        }

    terraform.tfvars
        server_details = ["web-server",3,true]

    main.tf
        resource "aws_instance" "web" {
            count =var.server_details[1]
            ami ="ami-123456"
            monitoring =var.server_details[2]
        }


8. Set
    Set is a collection of unique values.
    
    Difference from list: Allows duplicates.
        List:["dev","prod","dev"]

    Set:["dev","prod"] Duplicate removed.

    variables.tf
        variable "security_groups" {
            type = set(string)
        }

    terraform.tfvars
        security_groups = [
            "web-sg",
            "app-sg",
            "db-sg"
        ]

    main.tf
        Example security group rules:

        resource "aws_security_group" "example" {
            name = "app-sg"
            dynamic "ingress" {
                for_each = var.security_groups
                content {
                    description =ingress.value
                    from_port = 80
                    to_port = 80
                    protocol = "tcp"
                }
            }
        }
